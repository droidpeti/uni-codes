When we allow for elements to be equal, that is a binary sort tree, not a search tree

tree *search_tree(tree *l, item_type x){
    if(l == NULL){
        return NULL;
    }

    if(l->item == x){
        return l;
    }
    if(x < l->item){
        return search_tree(l->left, x);
    }
    else{
        return search_tree(l->right, x);
    }
}

3.

One way linked list cut list in at element number n, the first half will stay in L, return the second half

node* out(node *L; unsigned int n){
    node *p = L;
    unsigned int i = 1;

    while(i < n){
        i++;
        p = p->next;
    }

    node *q = p->next;
    p->next = NULL;
    return q;
}

void iter(tree* t){

void iter(tree* t){
    if(t == NULL){
        return;
    }

    iter(t->left);
    print(root(t));
    iter(t->right);
}

//post
void iter(tree* t){
    if(t == NULL){
        return;
    }

    iter(t->left);
    iter(t->right);
    print(root(t));
